136 
137  // Get original adjacncy matrix, θ(N^2)
138  initialize_adjacency_matrix(graph.edges, A, n);
139 
140  if (print_steps == true) {
141  printf("A_%i\n", 0);

145  //Floyd's Algorithm, θ(N^3)
146  for (int k = 0; k < n; k++) {
147  for (int i = 0; i < n; i++) {
148  for (int j = 0; j < n; j++) {
149  if (A[i][j] > A[i][k] + A[k][j]) {

170  /*
171  * Provided a list of edges, this function returns the original adjacency matrix
172  * that will be used with Floyd's
173  */
174  static void initialize_adjacency_matrix(std::vector<Edge>& edges, std::vector< std::vector<int> >& A, int n) {
175 
176  /*
177  Initialize adjacency matrix A with original path values
178  Loop through all indicies in the and see if there is an edge that connects them
179 
180  θ(N^2)
181  */
182  for (int i = 0; i < n; i++) {
183  for (int j = i; j < n; j++) {
184 
185  // If we are going down the diagnol, the distance from
186  // a vertex from and to itself is 0
187  if (i == j) {
188  A[i][j] = 0;
189 
190  // If i and j are not equal, then we have two verticies that are different
191  // and we need to find the weight of the edge that connects them. If there is
192  // no edge, the value will come back as infinity
193  } else {
194  // Make temporary verticies for values i and j, we will check if an edge connects them
195  Vertex u(i);
196  Vertex v(j);
197 
198  // Find the index of the edge that connects the verticies i and j
199  int index = Edge::find(edges, u, v);
200  int weight = INT_MAX;
201 
202  // If an edge was found, change the weight, otherwise leave it as infinity
203  // to represent there is no edge between verticies i and j
204  if (index != -1) {
205  weight = edges[index].weight;
206  }
207 
208  // Set the weight of the located edge to the corresponding locations
209  A[i][j] = weight;
210  A[j][i] = weight;
211  }
212  }
213  }
214  }
